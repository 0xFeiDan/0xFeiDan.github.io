    5. жһҡдёҫ enum

    жһҡдёҫпјҲenumпјүжҳҜsolidityдёӯз”ЁжҲ·е®ҡд№үзҡ„ж•°жҚ®зұ»еһӢгҖӮе®ғдё»иҰҒз”ЁдәҺдёәuintеҲҶй…ҚеҗҚз§°пјҢдҪҝзЁӢеәҸжҳ“дәҺйҳ…иҜ»е’Ңз»ҙжҠӨ

    enumзҡ„дёҖдёӘжҜ”иҫғеҶ·й—Ёзҡ„еҸҳйҮҸпјҢеҮ д№ҺжІЎд»Җд№Ҳдәәз”ЁгҖӮ

        // з”Ёenumе°Ҷuint 0пјҢ 1пјҢ 2иЎЁзӨәдёәBuy, Hold, Sell
    enum ActionSet { Buy, Hold, Sell }
    // еҲӣе»әenumеҸҳйҮҸ action
    ActionSet action = ActionSet.Buy;

    pragma solidity ^0.8.4; 

contract HelloWeb3{

    // з”Ёenumе°Ҷuint 0пјҢ 1пјҢ 2иЎЁзӨәдёәBuy, Hold, Sell
    enum ActionSet { Buy, Hold, Sell }
    // еҲӣе»әenumеҸҳйҮҸ action
    ActionSet action = ActionSet.Hold;


    // enumеҸҜд»Ҙе’ҢuintжҳҫејҸзҡ„иҪ¬жҚў  иҝ”еӣһActionSet.Hold = 1
    function enumToUint() external view returns(uint){
        return uint(action);
    }
}
